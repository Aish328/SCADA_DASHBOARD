# requirements.txt — feeder NHITS load-forecasting project
# Tested combination for Windows / Python 3.10 (64-bit), CPU-only.
#
# Install (inside your activated venv):
#     pip install -r requirements.txt
#
# The extra index below lets pip fetch the CPU-only torch wheel directly
# from the official PyTorch repository.
--extra-index-url https://download.pytorch.org/whl/cpu

# --- core numerical stack ---------------------------------------------------
# numpy stays < 2: avoids the numpy-2 ABI mismatch that caused silent
# access-violation crashes (0xC0000005) with older torch on Windows.
numpy<2                  # resolves to 1.26.4
pandas>=2.0,<3              # 2.x line is the safe pairing with numpy<2
scikit-learn>=1.3,<1.6

# --- deep learning ------------------------------------------------------------
# torch 2.4.1+cpu: neuralforecast 3.1.x requires torch>=2.4, and the
# newest wheels (2.12.x) failed DLL initialisation (WinError 1114 on
# c10.dll) on the Windows machine. 2.4.1 satisfies both. The "+cpu"
# local tag (resolved via the extra index above) keeps the install small
# and avoids pulling the nvidia-* CUDA packages on Linux.
torch==2.4.1+cpu
neuralforecast>=3.1,<4      # code tested on 3.1.9 (requires torch>=2.4)
pytorch-lightning>=2.0,<2.5

# --- plotting -----------------------------------------------------------------
matplotlib>=3.7,<3.10
